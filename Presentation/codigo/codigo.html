            ; this sample gets a string from a user, then it prints it out.
; this example uses dos interrupt: 21h

name "int21"

org  100h

jmp start


; when no file path is specified the default file path is:
;   c:\emu8086\MyBuild

; if path is specified, it is emulated in:
;   c:\emu8086\vdrive\
; for example, c:\test.txt is c:\emu8086\vdrive\c\test.txt
; this is done to prevent disasters and accidental overwriting of valuable files.
; relative paths such as "c:\mydir\..\autoexec.bat" are disabled for the same reasons.


; declarations:

; buffers hold ascii chars for INT 21h/0Ah input function,
; first byte is its maximum size, second byte: actual size.

; format of INT 21h/0Ah input buffer:
;  -byte-offset-     -number-of-bytes-             -description- 
;      00                  1                        buffer size
;      01                  1                        number of characters actually read (excluding cr).
;      02             1 to buffer-size              actual buffer contents (including cr). 

s1        db 100,?, 100 dup(' ') 
t2        db 100,?, 100 dup(' ')
filename  db 30,?, 30 dup(' ')   ; file name should be in 8.3 - dos compatible format.


; file handle:
handle   dw   0


; set segment registers to code:
start:  mov ax, cs
        mov ds, ax
        mov es, ax

jmp m1
msg1    db  "enter any text: $"
m1:     mov dx, offset msg1
        mov ah, 9
        int 21h
; input a string:
        mov dx, offset s1
        mov ah, 0ah
        int 21h



jmp m2
msg2    db  0Dh,0Ah,"enter file name: $"
m2:     mov dx, offset msg2
        mov ah, 9
        int 21h
; input filename:
        mov dx, offset filename
        mov ah, 0ah
        int 21h

; set 0 to the end of the filename:
        xor bx, bx
        mov bl, filename[1]  ; get actual size.
        mov filename[bx+2], 0

; create new file:
        mov cx, 0
        mov ah, 3ch
        mov dx, offset filename+2
        int 21h
        jc  error
        mov handle, ax
; write buffer to file:
        mov ah, 40h
        mov bx, handle        
        mov dx, offset s1+2
        xor cx, cx
        mov cl, s1[1]
        int 21h
        jc  error
; close file
        mov bx, handle
        mov ah, 3eh
        int 21h
        jc error
        
        
        
; open existing file:
        mov al, 0     ; read.
        mov ah, 3dh
        mov dx, offset filename+2
        int 21h
        jc  error
        mov handle, ax       
; read bytes from file
        mov ah, 3fh
        mov bx, handle
        xor cx, cx
        mov cl, t2[0]
        mov dx, offset t2[2]
        int 21h
        jc error
        mov t2[1], al
; close file
        mov bx, handle
        mov ah, 3eh
        int 21h
        jc error    
jmp m3
msg3    db  0Dh,0Ah,"read from file: $"
m3:     mov dx, offset msg3
        mov ah, 9
        int 21h
; set '$' to the end of the buffer:
        xor bx, bx
        mov bl, t2[1]        ; get actual size.
        mov t2[bx+2], '$'
; print the buffer:
        mov dx, offset t2[2] ; skip 2 control bytes.
        mov ah, 9
        int 21h
        
        
        jmp ok

; print error message:
error:  jmp m5
        msg5    db  0Dh,0Ah,"error...",0Dh,0Ah,'$'
        m5:     mov dx, offset msg5
                mov ah, 9
                int 21h


ok:     
jmp m4
msg4    db  0Dh,0Ah,"press any key...",0Dh,0Ah,'$'
m4:     mov dx, offset msg4
        mov ah, 9
        int 21h        
        mov ah, 0
        int 16h


; exit to the operating system:
        mov ah, 4ch
        int 21h

Interrupciones

INT 21h / AH = 0Ah

->

Recibe el input de una cadena de chars

<-

Lo almacen en DS:DX

Se debe a침adir manualmente $

Interrupciones

INT 21h / AH = 9h

->

Imprime una cadena de chars, debe de terminar en $

<-

Imprime lo almacenado en DS:DX

Interrupciones

INT 21h / AH = 3Ch

->

Crea o Trunca un archivo

<-

Input:

  • CX = Attributos de archivos.
    •             
                mov cx, 0       ;  normal - no attributes. 
 mov cx, 1       ;  read-only. 
 mov cx, 2       ;  hidden. 
 mov cx, 4       ;  system 
 mov cx, 7       ;  hidden, system and read-only! 
 mov cx, 16      ;  archive
  • DS:DX = Almacen el nombre en ASCII.

Cambia la bandera CF (Carry Flag) si fue exitoso

En AX garda el file handle o el c칩digo de error

Interrupciones

INT 21h / AH = 40h

->

Escribe en un archivo

<-

Input:

  • BX = File handle.
  • CX = cantidad de Bytes a escribir.
  • DS:DX = Datos a escribir.

Cambia la bandera CF (Carry Flag) si fue exitoso

En AX guarda los bytes escritos o el c칩digo de error

<-

Input:

  • BX = File handle.
  • CX = cantidad de Bytes a leer.
  • DS:DX = Buffer para datos.

Cambia la bandera CF (Carry Flag) si fue exitoso

            org 100h  ; Programa .COM

            .model small
            .stack 100h
            
            .data
            menu db '1. Linea',13,10
                 db '2. Rectangulo',13,10
                 db '3. Triangulo',13,10
                 db '4. Limpiar',13,10
                 db '5. Salir',13,10, '$'
            
            .code
            
            inicio:
                call EstablecerModoVideo13h
            bucle_principal:
                call MostrarMenu
                call EsperarTecla
                cmp al, '1'
                je DibujarLinea
                cmp al, '2'
                je DibujarRectangulo
                cmp al, '3'
                je DibujarTriangulo
                cmp al, '4'
                je LimpiarPantalla
                cmp al, '5'
                je SalirPrograma
                jmp bucle_principal
            
            DibujarLinea:
                ; dibujar l칤nea horizontal de (50,50) a (150,50)
                mov cx, 50
            .siguiente_pixel:
                cmp cx, 150
                ja .hecho
                mov dx, 50
                call PonerPixel
                inc cx
                jmp .siguiente_pixel
            .hecho:
                jmp bucle_principal
             
             
             
            DibujarRectangulo:
                ; l칤nea superior
                mov cx, 100
            .dibujar_superior:
                cmp cx, 200
                ja .lateral_izquierdo
                mov dx, 60
                call PonerPixel
                inc cx
                jmp .dibujar_superior
            
            .lateral_izquierdo:
                mov dx, 60
            .dibujar_lado_izquierdo:
                cmp dx, 100
                ja .lateral_derecho
                mov cx, 100
                call PonerPixel
                inc dx
                jmp .dibujar_lado_izquierdo
            
            .lateral_derecho:
                mov dx, 60
            .dibujar_lado_derecho:
                cmp dx, 100
                ja .dibujar_inferior
                mov cx, 200
                call PonerPixel
                inc dx
                jmp .dibujar_lado_derecho
            
            .dibujar_inferior:
                mov cx, 100
            .dibujar_base:
                cmp cx, 200
                ja .fin_rect
                mov dx, 100
                call PonerPixel
                inc cx
                jmp .dibujar_base
            .fin_rect:
                jmp bucle_principal
            
            
            
            
            
            DibujarTriangulo:
                ; Tri치ngulo hacia abajo con v칠rtice en (125, 50) y base en Y=100
                mov dx, 50          ; Y inicial (v칠rtice)
            .triangulo_loop:
                cmp dx, 100         ; hasta Y=100
                ja .fin_triangulo
            
                ; calcular extremos izquierdo y derecho de la l칤nea horizontal actual
                mov ax, dx
                sub ax, 50          ; altura desde la cima
                mov bx, 125
                sub bx, ax          ; X izquierdo = 125 - altura
                mov cx, bx          ; CX = X izquierdo
            
                ; X derecho = 125 + altura (guardamos en BX)
                mov bx, 125
                add bx, ax
            
            .dibujar_linea_horizontal:
                cmp cx, bx
                ja .siguiente_linea
                call PonerPixel
                inc cx
                jmp .dibujar_linea_horizontal
            
            .siguiente_linea:
                inc dx
                jmp .triangulo_loop
            
            .fin_triangulo:
                jmp bucle_principal
            
            
            LimpiarPantalla:
                call EstablecerModoVideo13h
                jmp bucle_principal
            
            MostrarMenu:
                mov ah, 09h
                lea dx, menu
                int 21h
                ret
            
            EsperarTecla:
                mov ah, 00h
                int 16h
                ret
            
            PonerPixel:
                ; entrada: CX = X, DX = Y
                ; usa: AL = color (por defecto = 15)
                push ax
                push bx
                push cx
                push dx
                mov al, 15
                mov ah, 0Ch
                mov bh, 0
                int 10h
                pop dx
                pop cx
                pop bx
                pop ax
                ret
            
            EstablecerModoVideo13h:
                mov ah, 0
                mov al, 13h
                int 10h
                ret
            
            SalirPrograma:
                ; volver al modo texto antes de salir
                mov ax, 0003h
                int 10h
                mov ah, 4Ch
                int 21h
            end inicio
